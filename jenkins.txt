pipeline {
    agent any
    stages {
        stage('Hello') {
            steps {
                echo 'Hello World'
                echo 'pipeline successfully created'
                 }
             }
        stage('Build') {
             steps {
                sh 'mvn clean compile'  // If using Gradle, use 'gradle build'
                 }
             }

        stage('Run Cucumber Tests') {
            steps {
                sh 'mvn test'  // Runs Cucumber tests
            }
        }
    }
}
